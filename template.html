def baz_the_foo(
    n:   int|str,  # Move this many
    old: list,     # From this stack
    new: list,     # To this stack
    )-> None:      # MODIFIES the stacks!
    """Move `n` things from `from_col` to `to_col` AS A GROUP."""
    for x in reversed([old.pop() for i in range(int(n))]):
        new.append(x)
